 124    def setImages(self, filenames):
 125        """Set the list of image filenames.
 126
 127        Args:
 128            filenames (list[str]): List of image filenames
 129        """
 130        if pref.verbose: print(" [MODEL] >> ImageGalleryModel.setImages(", len(list(copy.deepcopy(filenames))), "images)")
 131        self.imageFilenames = list(filenames)
 132        self.imagesMetadata, self.processPipes = [], [] # reset metadata and processPipes
 133        self.aestheticsModels = [] # reset aesthetics models
 134        nbImagePage = controller.GalleryMode.nbRow(self.controller.view.shapeMode) * controller.GalleryMode.nbCol(self.controller.view.shapeMode)
 135        for f in self.imageFilenames: # load only first page
 136            self.processPipes.append(None)
 137        self.controller.updateImages() # update controller to update view
 138        self.loadPage(0)
 139
 140    def loadPage(self, nb):
 141        """Load a specific page of images.
 142
 143        Args:
 144            nb (int): Page number
 145        """
 146        if pref.verbose: print(" [MODEL] >> ImageGalleryModel.loadPage(", nb, ")")
 147        nbImagePage = controller.GalleryMode.nbRow(self.controller.view.shapeMode) * controller.GalleryMode.nbCol(self.controller.view.shapeMode)
 148        min_, max_ = (nb * nbImagePage), ((nb + 1) * nbImagePage)
 149        loadThreads = thread.RequestLoadImage(self)
 150        for i, f in enumerate(self.imageFilenames[min_:max_]): # load only the current page nb
 151            if not isinstance(self.processPipes[min_ + i], hdrCore.processing.ProcessPipe):
 152                self.controller.parent.statusBar().showMessage("read image: " + f)
 153                self.controller.parent.statusBar().repaint()
 154                loadThreads.requestLoad(min_, i, f)
 155            else:
 156                self.controller.view.updateImage(i, self.processPipes[min_ + i], f)
 157
 158    def save(self):
 159        """Save the current state."""
 160        if pref.verbose: print(" [MODEL] >> ImageGalleryModel.save()")
 161        for i, p in enumerate(self.processPipes):
 162            if isinstance(p, hdrCore.processing.ProcessPipe):

 586                    maxValue = max(listValues[:index, 1])
